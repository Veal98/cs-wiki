class Solution {
    public int[] twoSum(int[] numbers, int target) {
        int len = numbers.length;
        int left = 0;
        int right = len - 1;
        int[] res = new int[2];
        while(left < right){
            int sum = numbers[left] + numbers[right];
            if(sum == target){
                // é¢˜æ„ä¸­ä¸‹æ ‡ä» 1 å¼€å§‹
                res[0] = left + 1;
                res[1] = right + 1;
                break;
            }
            else if(sum < target)
                left ++;
            else
                right --;    
        }

        return res;
    }
}
