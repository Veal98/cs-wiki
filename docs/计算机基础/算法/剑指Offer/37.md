public class Solution {
    public int GetNumberOfK(int [] array , int k) {
       if(array.length <= 0 || k < array[0] || k > array[array.length-1]){
            return 0;
        }
        int left = 0;
        int right = array.length - 1;
        int mid = -1;
        int count = 0; //è®°å½•å‡ºç°çš„æ¬¡æ•°

            mid = (left + right) / 2;

                left = mid + 1;
            else if(array[mid] > k)
                right = mid - 1;
            else if(array[mid] == k){
                count ++;
                break;
            }
        }
        int pre = mid - 1;
        int foll = mid + 1;
        while(pre >= left){
            if(array[pre] == k){
                count ++;
                pre --;
            }else
                break;
        }
        while(foll <= right){
            if(array[foll] == k){
                count ++;
                foll ++;
            }else
                break;
        }
        
        return count;
    }
}
