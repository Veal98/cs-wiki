class Solution {
public:
    ListNode* FindFirstCommonNode( ListNode* pHead1, ListNode* pHead2) {
        int a = 0,b = 0;
        int dif;
        ListNode* p = pHead1, *q = pHead2;
        if(pHead1 == NULL || pHead2 == NULL) return NULL;
        if(pHead1 == pHead2) return pHead1;
        while(p != NULL){
            a ++;
            p = p->next;
        }
        while(q != NULL){
            b ++;
            q = q->next;
        }
        p = pHead1;
        q = pHead2;
        if(a > b){
            dif = a - b;
            while(dif > 0){
                p = p->next;
                dif --;
            }
            while(p != q){
                p = p->next;
                q = q->next;
            }
            return p;
        }else{
            dif = b - a;
            while(dif > 0){
                q = q->next;
                dif --;
            }
            while(q != p){
                p = p->next;
                q = q->next;
            }
            return q;
        }
         
    }
};
