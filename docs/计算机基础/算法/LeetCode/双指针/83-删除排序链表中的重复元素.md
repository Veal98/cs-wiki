class Solution {
    public ListNode deleteDuplicates(ListNode head) {
        int slow = head;
        int fast = head;
        while (fast != null) {
            if (fast.val != slow.val) {
                // slow = fast
                slow.val = fast.val;
                // slow ++
                slow = slow.next;
            }
            // fast ++
            fast = fast.next;
        }
        // æ–­å¼€ä¸åé¢é‡å¤å…ƒç´ çš„è¿æ¥
        slow.next = null;
        return head;
    }
}
