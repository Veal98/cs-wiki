    public int findKthLargest(int[] nums, int k) {
        int len = nums.length;
        int left = 0;
        int right = len - 1;

    private int partition(int[] nums, int left, int right) {

        int r = right;
        
        int pivot = nums[left]; // é€‰å–æœ€å·¦è¾¹çš„æ•°ä¸ºä¸­è½´ç‚¹

            }
            
        }
        // swap(nums[l], nums[left])
        nums[left] = nums[l];
        nums[l] = pivot;
        
        return l;
    }
