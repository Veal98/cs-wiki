    public int maxDepth(TreeNode root) {
        if (root == null) {
            return 0;
        }

        Queue<TreeNode> queue = new LinkedList<>();
        // æ ¹èŠ‚ç‚¹å…¥é˜Ÿ

        while (!queue.isEmpty()) {

            int size = queue.size();
            depth ++;
            for (int i = 0;  i < size;  i++) {
                TreeNode cur = queue.poll();

                if (cur.left != null) {
                    queue.offer(cur.left);
                }
                if (cur.right != null) {
                    queue.offer(cur.right);
                }
            }
        }
        
        return depth;
    }
}
