class Solution {
    public int[] twoSum(int[] numbers, int target) {
        if (numbers == null || numbers.length == 0 || numbers[0] > target) {
            return new int[0];
        }

        int left = 0;
        int right = numbers.length - 1;

        while (left < right) {
            int sum = numbers[left] + numbers[right];
            if (sum == target) {
                return new int[]{left, right};
            }
            else if (sum < target) {
                left ++;
            }
            else {
                right --;
            }
        }

        return new int[0];
    }
}
