    public int nthUglyNumber(int n) {
        if (n == 1) {

        int a = 1, b = 1, c = 1;
        for (int i = 2; i <= n; i ++) {
            int n2 = dp[a] * 2;

            int n5 = dp[c] * 5;

    }
}
