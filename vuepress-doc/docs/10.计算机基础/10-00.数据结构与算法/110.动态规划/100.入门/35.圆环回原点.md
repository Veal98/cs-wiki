public class Solution {
        
    private static final int MOD = 1000000007;
    
    public int circle (int n) {
        int len = 10;
        int[][] dp = new int[n + 1][len];

        // base case
        dp[0][0] = 1;

        for (int i = 1; i <= n; i ++) {
            for (int j = 0; j < len; j ++) {
                dp[i][j] = (dp[i - 1][(j - 1 + len) % len] + dp[i - 1][(j + 1) % len]) % MOD;
            }
        }

        return dp[n][0];
    }
}
