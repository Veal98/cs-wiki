public class Solution {
    public boolean Find(int target, int [][] array) {
        
        for (int i = 0; i < array.length; i ++) {
            int left = 0;
            int right = array[0].length - 1;
            while (left <= right) {
                int mid = left + ((right - left) >> 1);
                if (array[i][mid] == target) {
                    return true;
                }
                else if (array[i][mid] > target) {
                    right = mid - 1;
                }
                else {
                    left = mid + 1;
                }
            }
        }
        
        return false;
    }
}
