class Solution {
    public int maxDistance(int[][] grid) {
        int row = grid.length;
        int col = grid[0].length;

        Queue<int[]> queue = new LinkedList<>();
        boolean[][] visited = new boolean[row][col];
        //  grid ä¸­å¯¹åº”ä½ç½®å…ƒç´ åˆ°æœ€è¿‘çš„ 1 çš„è·ç¦»
        int[][] dists = new int[row][col];

        // é™†åœ°èŠ‚ç‚¹ 1 å…¥é˜Ÿ
        for (int i = 0; i < row; i ++) {
            for (int j = 0; j < col; j ++) {
                if (grid[i][j] == 1) {
                    queue.offer(new int[]{i, j});
                    visited[i][j] = true;
                }
            }
        }

        if (queue.size() == 0 || queue.size() == row * col) {
            return -1;
        }

        int[][] directions = {{-1, 0}, {1, 0}, {0, 1}, {0, -1}};
        while (!queue.isEmpty()) {
            int[] cur = queue.poll();
            int dist = dists[cur[0]][cur[1]];

            for (int[] dir : directions) {
                int ni = cur[0] + dir[0];
                int nj = cur[1] + dir[1];

                if (ni >= 0 && ni < row && nj >= 0 && nj < col && grid[ni][nj] == 0 && !visited[ni][nj]) {
                    visited[ni][nj] = true;
                    queue.offer(new int[]{ni, nj});
                    // dist æ˜¯å½“å‰èŠ‚ç‚¹ cur åˆ°æœ€è¿‘çš„ 1 çš„è·ç¦»ï¼Œ
                    // ä»å½“å‰æ ¼å­ cur èµ°ä¸€æ­¥åˆ°è¾¾è¯¥å€¼ä¸º 0 çš„æ ¼å­(ni, nj)ï¼Œé‚£ä¹ˆè¯¥æ ¼å­ç¦»æœ€è¿‘çš„ 1 çš„è·ç¦»æ˜¯ dist + 1
                    dists[ni][nj] = dist + 1;
                }
            }
        }

        // æœ€çŸ­è·ç¦»ä¸­æ‰¾æœ€å¤§å€¼
        int res = 0;
        for (int i = 0; i < row; i ++) {
            for (int j = 0; j < col; j ++) {
                res = Math.max(res, dists[i][j]);
            }
        }

        return res;
    }
}
