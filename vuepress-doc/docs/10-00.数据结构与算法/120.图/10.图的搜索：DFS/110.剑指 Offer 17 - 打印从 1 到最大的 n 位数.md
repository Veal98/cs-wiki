class Solution {
    public int[] printNumbers(int n) {
        int end = 9;
        while (n != 1) {
            end = end * 10 + 9;
            n --;
        }

        List<Integer> list = new ArrayList<>();
        for (int i = 1; i <= end; i ++) {
            list.add(i);
        }

        return list.stream().mapToInt(Integer::intValue).toArray();
    }
}
