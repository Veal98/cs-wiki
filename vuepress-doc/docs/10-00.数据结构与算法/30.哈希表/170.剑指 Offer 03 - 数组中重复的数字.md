class Solution {
    public int findRepeatNumber(int[] nums) {
        int i = 0;
        while (i < nums.length) {
            int targetIndex = nums[i];
            // å½“å‰ nums[i] å‡ºç°åœ¨æ­£ç¡®çš„ä½ç½®
            if (targetIndex == i) {
                i ++;

            else if (nums[targetIndex] != nums[i]) {
                swap(nums, i, targetIndex);
            } 
            // nums[targetIndex] == nums[i], é‡å¤æ•°å­—
            else {
                return nums[i];
            }
        }

        return -1;
    }

    private void swap(int[] nums, int i, int j) {
        int temp = nums[i];
        nums[i] = nums[j];
        nums[j] = temp;
    }
}
