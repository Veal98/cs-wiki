import java.util.*;

public class Solution {
    public int[][] rotateMatrix(int[][] mat, int n) {
        // 1. è½¬ç½®
        for (int i = 0; i < n; i ++) {
            for (int j = 0; j < i; j ++) {
                int temp = mat[i][j];
                mat[i][j] = mat[j][i];
                mat[j][i] = temp;
            }
        }
        
        // 2. æ¯è¡Œ reverse
        for (int[] row : mat) {
            reverse(row, 0, n - 1);
        }
        
        return mat;
    }
    
    private void reverse(int[] a, int left, int right) {
        while (left < right) {
            int temp = a[left];
            a[left] = a[right];
            a[right] = temp;
            
            left ++;
            right --;
        }
    }
}
