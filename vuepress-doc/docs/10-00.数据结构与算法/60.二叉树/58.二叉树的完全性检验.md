class Solution {
    public boolean isCompleteTree(TreeNode root) {
        if (root == null || (root.left == null && root.right == null)) {
            return true;
        }

        Queue<TreeNode> queue = new LinkedList<>();
        queue.offer(root);

        while (!queue.isEmpty()) {
            TreeNode cur = queue.poll();
            // é‡åˆ°ç©ºç»“ç‚¹ä¹‹åå†é‡åˆ°ä¸ä¸ºç©ºç»“ç‚¹çš„ç»“ç‚¹ï¼Œè¯´æ˜ä¸ä¸ºå®Œå…¨äºŒå‰æ ‘
            if (cur == null && !queue.isEmpty() && queue.peek() != null) {
                return false;
            }

            if (cur != null) {
                // null ä¹Ÿè¦å…¥é˜Ÿï¼Œæ‰€ä»¥ä¸è¦åŠ  cur.left != null çš„åˆ¤æ–­

        }

        return true;
    }
}
