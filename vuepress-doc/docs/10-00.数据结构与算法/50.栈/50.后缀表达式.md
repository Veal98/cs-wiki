class Solution {
    public int evalRPN(String[] tokens) {
        if (tokens == null || tokens.length == 0) {
            return 0;
        }

        Stack<Integer> stack = new Stack<>();

        for (String s : tokens) {
            if (s.equals("+") || s.equals("-") || s.equals("*") || s.equals("/")) {
                int op1 = stack.pop();
                int op2 = stack.pop();
                int res = 0;
                switch (s) {
                    case "+":
                        res = op2 + op1;
                        break;
                    case "-":
                        res = op2 - op1;
                        break;
                    case "*":
                        res = op2 * op1;
                        break;
                    case "/":
                        res = op2 / op1;
                        break;
                }
                stack.push(res);
            } else {
                stack.push(new Integer(s));
            }
        }

        // æ ˆä¸­çš„æœ€åä¸€ä¸ªå…ƒç´ å°±æ˜¯æœ€ç»ˆç»“æœ
        return stack.pop();
