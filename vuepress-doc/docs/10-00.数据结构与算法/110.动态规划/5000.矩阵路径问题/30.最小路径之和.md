    public int minPathSum(int[][] grid) {
        int row = grid.length;

        // dp[i][j]: ä» (0,0) åˆ° (i,j) çš„æœ€å°è·¯å¾„å’Œ

        }
		// è®¡ç®—ç¬¬ä¸€è¡Œçš„æœ€å°è·¯å¾„å’Œ
        sum = 0;
        for (int j = 0; j < col; j ++) {
            sum += grid[0][j];
            dp[0][j] = sum;
        }

        // loop
        for (int i = 1; i < row; i ++) {
            for (int j = 1; j < col; j ++) {

            }
        }

        return dp[row - 1][col - 1];
    }
}
