class Solution {
    public boolean isSubsequence(String s, String t) {
        if (s == null || s.length() == 0) {
            return true;
        }

        int len1 = s.length();
        int len2 = t.length();
        boolean[][] dp = new boolean[len1 + 1][len2 + 1];

        // base case
        for (int j = 0; j <= len2; j ++) {
            dp[0][j] = true;
        }

        // loop
        for (int i = 1; i <= len1; i ++) {
            for (int j = 1; j <= len2; j ++) {
                if (s.charAt(i - 1) == t.charAt(j - 1)) {
                    dp[i][j] = dp[i - 1][j - 1];
                }
                else {
                    dp[i][j] = dp[i][j - 1];
                }
            }
        }

        return dp[len1][len2];
    }
}
