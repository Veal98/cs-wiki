```java
class Solution {
    public int findLengthOfLCIS(int[] nums) {
        if (nums == null || nums.length == 0) {
            return 0;
        }

        int[] dp = new int[nums.length];

        // base case
        Arrays.fill(dp, 1);
		
        // æœ€ç»ˆç»“æœ
        int res = dp[0];
        
        // loop
        for (int i = 1; i < dp.length; i ++) {
            if (nums[i] > nums[i - 1]) {
                dp[i] = dp[i - 1] + 1;
            }
            res = Math.max(dp[i], res);
        }

        return res;
    }
}
```
