class Solution {
    public void merge(int[] nums1, int m, int[] nums2, int n) {
        if (nums2 == null || nums2.length == 0) {
            return ;
        }

        int[] nums3 = new int[m + n];

        int p1 = 0;
        int p2 = 0;
        int p3 = 0;

        while (p1 < m && p2 < n) {
            if (nums1[p1] < nums2[p2]) {
                nums3[p3 ++] = nums1[p1 ++];
            }
            else {
                nums3[p3 ++] = nums2[p2 ++];
            }
        }

        while (p1 < m) {
            nums3[p3 ++] = nums1[p1 ++];
        }
        while (p2 < n) {
            nums3[p3 ++] = nums2[p2 ++];
        }

        for (int i = 0; i < nums1.length; i ++) {
            nums1[i] = nums3[i];
        }
        
    }
}
