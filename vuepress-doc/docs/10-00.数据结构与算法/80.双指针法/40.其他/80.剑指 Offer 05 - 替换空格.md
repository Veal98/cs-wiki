class Solution {
    public String replaceSpace(String s) {
        if (s == null || s.length() == 0) {
            return s; 
        }
		
        // ç©ºæ ¼çš„ä¸ªæ•°
        int count = 0;
        for (char c : s.toCharArray()) {
            if (c == ' ') {
                count ++;
            }
        }

        char[] res = new char[s.length() + count * 2];
        for (int i = 0; i < s.length(); i ++) {
            res[i] = s.charAt(i);
        }

        int i = s.length() - 1;
        int j = res.length - 1;

        while (i >= 0) {
            if (s.charAt(i) == ' ') {
                res[j --] = '0';
                res[j --] = '2';
                res[j --] = '%';
            }
            else {
                res[j --] = s.charAt(i);
            }

            i --;
        }

        return new String(res);
    }
}
