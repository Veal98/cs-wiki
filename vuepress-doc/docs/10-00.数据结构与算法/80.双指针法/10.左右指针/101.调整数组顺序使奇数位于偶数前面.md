class Solution {
    public int[] exchange(int[] nums) {
        int left = 0;
        int right = left + 1;
        while (left < nums.length) {
            // å½“å‰æ•°æ˜¯å¶æ•°ï¼Œå‘åæ‰¾åˆ°ç¬¬ä¸€ä¸ªå¥‡æ•°ä¸å…¶äº¤æ¢
            if (nums[left] % 2 == 0) {
                right = left + 1;
                for (; right < nums.length; right ++) {
                    if (nums[right] % 2 != 0) {
                        swap(nums, left, right);
                        break;
                    }
                }
            }

            left ++;
        }

        return nums;
    }

    private void swap(int[] nums, int left, int right) {
        int temp = nums[left];
        nums[left] = nums[right];
        nums[right] = temp;
    }
}
