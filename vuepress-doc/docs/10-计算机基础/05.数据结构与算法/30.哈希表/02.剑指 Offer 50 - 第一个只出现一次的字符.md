class Solution {
    public char firstUniqChar(String s) {
        Map<Character, Integer> map = new HashMap<>();
        for (Character c : s.toCharArray()) {
            map.put(c, map.getOrDefault(c, 0) + 1);
        }

        for (Character c : s.toCharArray()) {
            if (map.get(c) == 1) {
                return c;
            }
        }

        return ' ';
    }
}
