class Solution {
    public TreeNode insertIntoBST(TreeNode root, int val) {
        if (root == null) {
            return new TreeNode(val);
        }

        if (root.val > val) {
            if (root.left == null) {
                root.left = new TreeNode(val);
            }
            else {
                insertIntoBST(root.left, val);
            }
        }

        else if (root.val < val) {
            if (root.right == null) {
                root.right = new TreeNode(val);
            }
            else {
                insertIntoBST(root.right, val);
            }
        }

        return root;
        
    }
}

class Solution {
    public TreeNode insertIntoBST(TreeNode root, int val) {
        if (root == null) {
            return new TreeNode(val);
        }

        TreeNode cur = root;
        while (cur != null) {
            // å·¦
            if (cur.val > val) {
                if (cur.left == null) {
                    cur.left = new TreeNode(val);
                    break;
                } else {
                    cur = cur.left;
                }
            }
            // å³
            if (cur.val < val) {
                if (cur.right == null) {
                    cur.right = new TreeNode(val);
                    break;
                } else {
                    cur = cur.right;
                }
            }
        }

        return root;
    }
}
